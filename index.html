def generate_geomancy_chart(m1, m2, m3, m4): """ Kàbbalu ràmmu: Boo boolee poñ yi: Sum bu mat (even) -> 2 poñ Sum bu manke (odd) -> 1 poñ """ def add_figures(f1, f2): return [2 if (x + y) % 2 == 0 else 1 for x, y in zip(f1, f2)] # 1. Tàbbal 4 Ndiy yi (Mothers) M = {1: m1, 2: m2, 3: m3, 4: m4} # 2. Sàkk 4 Doom yi (Daughters: M5 ba M8) ci sàllu Ndiy yi M[5] = [M[1][0], M[2][0], M[3][0], M[4][0]] M[6] = [M[1][1], M[2][1], M[3][1], M[4][1]] M[7] = [M[1][2], M[2][2], M[3][2], M[4][2]] M[8] = [M[1][3], M[2][3], M[3][3], M[4][3]] # 3. Sàkk 4 Sëfët yi (Nephews: M9 ba M12) M[9] = add_figures(M[1], M[2]) # Këru Tukki M[10] = add_figures(M[3], M[4]) # Këru Liggéey M[11] = add_figures(M[5], M[6]) # Këru Yaakaar M[12] = add_figures(M[7], M[8]) # Këru Noon # 4. Sàkk Seere yi ak Àtte bi (M13 ba M15) M[13] = add_figures(M[9], M[10]) # Ki togg M[14] = add_figures(M[11], M[12]) # Xabar buy ñëw M[15] = add_figures(M[13], M[14]) # ÀTTE BI (Haliyyu / Ousmane) # 5. Sàkk Firndéel gu mujj gi (M16 = M15 + M1) M[16] = add_figures(M[15], M[1]) return M # --- 🚀 REEFEERE REZULTO CI SA 4 YONENT YI --- # Solaymaan= [1, 1, 2, 1] | Ayyuub= [2, 1, 1, 1] | Muhammad= [1, 2, 1, 2] | Yuusufa= [2, 2, 1, 1] jant_bi = generate_geomancy_chart([1, 1, 2, 1], [2, 1, 1, 1], [1, 2, 1, 2], [2, 2, 1, 1]) print("Figiir bu Kër 15 (Àtte bi) :", jant_bi[15]) # Dina genn [1, 1, 1, 1] buy woné Adama print("Figiir bu Kër 16 (Firndéel):", jant_bi[16]) # Dina genn [2, 2, 1, 2]
